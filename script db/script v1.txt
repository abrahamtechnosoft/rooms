-- ============================================================
-- TABLA 1: Asistencia de participantes
-- ============================================================
-- Marca si cada participante asistio o no, con motivo opcional.
-- Se puede marcar durante la reunion o hasta 24h despues.

CREATE TABLE core.reservation_attendance (
    attendance_id INT IDENTITY(1,1) PRIMARY KEY,
    reservation_id INT NOT NULL,
    user_id INT NOT NULL,
    attended BIT NOT NULL,                       -- 1 = asistio, 0 = no asistio
    absence_reason VARCHAR(300) NULL,            -- motivo si no asistio (opcional)
    marked_by INT NOT NULL,                      -- quien marco la asistencia (organizador o admin)
    marked_at DATETIME2 NOT NULL DEFAULT SYSDATETIME(),
    
    CONSTRAINT FK_attendance_reservation 
        FOREIGN KEY (reservation_id) REFERENCES core.reservations(reservation_id),
    CONSTRAINT FK_attendance_user 
        FOREIGN KEY (user_id) REFERENCES auth.users(user_id),
    CONSTRAINT FK_attendance_marker 
        FOREIGN KEY (marked_by) REFERENCES auth.users(user_id),
    CONSTRAINT UQ_attendance_reservation_user 
        UNIQUE (reservation_id, user_id)
);
GO

-- ============================================================
-- TABLA 2: Notas generales de la reunion

-- Tabla de bloqueos personales del usuario
CREATE TABLE auth.user_blocks (
    block_id INT IDENTITY(1,1) PRIMARY KEY,
    user_id INT NOT NULL,
    name VARCHAR(80) NOT NULL,                  -- "Almuerzo", "Incapacidad", "Focus time"
    block_type VARCHAR(20) NOT NULL,            -- 'recurring' | 'one_time'
    
    -- Para recurrentes:
    days_of_week VARCHAR(30) NULL,              -- '1,2,3,4,5' = lun-vie
    start_time TIME NULL,                       -- 12:00
    end_time TIME NULL,                         -- 13:00
    
    -- Para puntuales (rango de fechas):
    start_at DATETIME2 NULL,
    end_at DATETIME2 NULL,
    
    is_active BIT NOT NULL DEFAULT 1,
    created_at DATETIME2 NOT NULL DEFAULT SYSDATETIME(),
    
    CONSTRAINT FK_user_blocks_user 
        FOREIGN KEY (user_id) REFERENCES auth.users(user_id),
    CONSTRAINT CK_user_blocks_type 
        CHECK (block_type IN ('recurring', 'one_time')),
    CONSTRAINT CK_user_blocks_recurring_fields 
        CHECK (
            (block_type = 'recurring' AND days_of_week IS NOT NULL 
                AND start_time IS NOT NULL AND end_time IS NOT NULL)
            OR
            (block_type = 'one_time' AND start_at IS NOT NULL AND end_at IS NOT NULL)
        )
);
GO

CREATE INDEX IX_user_blocks_user_active 
    ON auth.user_blocks(user_id, is_active);
GO

CREATE TABLE core.recurring_series (
    series_id INT IDENTITY(1,1) PRIMARY KEY,

);
